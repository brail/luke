## 💻 Backend Implementation

### exceljs Template

```typescript
import ExcelJS from 'exceljs';

const COLUMN_HEADERS = [
  'CustomerCheckCode',
  'CustomerCheckName',
  'SalesPersonCheckCode',
  'SalesPersonCheckName',
  'Document No_',
  'Line No_',
  // ... tutte le 232 colonne nell'ordine esatto
  'Authorized Stores Number'
];

const COLUMN_FORMATS = {
  'ValueSold': '€ #,##0.00',
  'ValueShipped': '€ #,##0.00',
  'ValueInvoiced': '€ #,##0.00',
  // ... tutte le colonne di valuta
  'Order Date': 'DD/MM/YYYY',
  'Delete Date': 'DD/MM/YYYY',
  // ... tutte le date
  'QuantitySold': '#,##0',
  'PairsSold': '#,##0',
  // ... tutti i numeri
};

export async function generateAnalisiVendite(rows: any[]) {
  const workbook = new ExcelJS.Workbook();
  const worksheet = workbook.addWorksheet('AnalisiVendite');

  // Setup columns
  worksheet.columns = COLUMN_HEADERS.map((header, index) => ({
    header,
    key: header.toLowerCase().replace(/ /g, '_'),
    width: 12, // Auto-adjust in Excel
  }));

  // Format header
  const headerRow = worksheet.getRow(1);
  headerRow.fill = { type: 'pattern', pattern: 'solid', fgColor: { argb: 'FF003366' } };
  headerRow.font = { bold: true, color: { argb: 'FFFFFFFF' }, size: 11 };
  headerRow.alignment = { horizontal: 'center', vertical: 'center', wrapText: true };

  // Freeze panes & AutoFilter
  worksheet.freezePane = 'A2';
  worksheet.autoFilter.from = 'A1';
  worksheet.autoFilter.to = `${getColumnLetter(232)}1`;

  // Add data rows
  let rowNum = 2;
  for (const row of rows) {
    const excelRow = worksheet.addRow(row);

    // Alternating colors
    if (rowNum % 2 === 0) {
      excelRow.fill = { type: 'pattern', pattern: 'solid', fgColor: { argb: 'FFF2F2F2' } };
    }

    // Apply formatting to each cell
    for (const [colIndex, header] of COLUMN_HEADERS.entries()) {
      const cell = excelRow.getCell(colIndex + 1);

      if (COLUMN_FORMATS[header]) {
        cell.numFmt = COLUMN_FORMATS[header];
      }

      // Right align currency/numbers
      if (header.includes('Value') || header.includes('Quantity') || header.includes('Pairs') || header.includes('Cost') || header.includes('Margin') || header.includes('Commission') || header.includes('Sconto') || header.includes('Provvigione')) {
        cell.alignment = { horizontal: 'right' };
      }
    }

    rowNum++;
  }

  return workbook.xlsx.writeBuffer();
}
```

### tRPC Endpoint

```typescript
export const analysisRouter = router({
  exportAnalisiVendite: publicProcedure
    .input(z.object({
      dateFrom: z.date(),
      dateTo: z.date(),
      filters: z.object({
        seasons: z.array(z.string()).optional(),
        customers: z.array(z.string()).optional(),
        salespersons: z.array(z.string()).optional(),
      }).optional(),
    }))
    .mutation(async ({ input }) => {
      // Query: udf_AnalisiVenditeComplete with where clause
      const rows = await db.query(
        `SELECT * FROM udf_AnalisiVenditeComplete()
         WHERE [Order Date] BETWEEN @dateFrom AND @dateTo
         ${input.filters?.seasons ? 'AND [Season Code] IN (@seasons)' : ''}
         ${input.filters?.customers ? 'AND [Sell-to Customer No_] IN (@customers)' : ''}
         ${input.filters?.salespersons ? 'AND [Salesperson Code] IN (@salespersons)' : ''}`,
        {
          dateFrom: input.dateFrom,
          dateTo: input.dateTo,
          seasons: input.filters?.seasons || null,
          customers: input.filters?.customers || null,
          salespersons: input.filters?.salespersons || null,
        }
      );

      // Generate Excel
      const buffer = await excelService.generateAnalisiVendite(rows);

      return {
        success: true,
        buffer,
        filename: `AnalisiVendite_${format(input.dateTo, 'yyyy-MM-dd')}.xlsx`,
        rowCount: rows.length,
      };
    }),
});
```
